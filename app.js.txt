// ==================== GAME STATE ====================
const gameState = {
    level: 1,
    xp: 0,
    xpToNext: 100,
    currentQuestions: [],
    currentQuestionIndex: 0,
    score: 0,
    wrongAnswers: [],
    timerInterval: null,
    timeLeft: 0,
    timerLimit: 0,
    answered: false,
    config: {
        focus: '',
        area: '',
        questionCount: 5,
        timerLimit: 0
    }
};

const levels = [
    { name: "Concurseiro Iniciante", xpNeeded: 100 },
    { name: "Estudante Dedicado", xpNeeded: 250 },
    { name: "Concurseiro Intermediario", xpNeeded: 500 },
    { name: "Expert em Provas", xpNeeded: 800 },
    { name: "Mestre dos Concursos", xpNeeded: 1200 },
    { name: "Oraculo da Aprovacao", xpNeeded: 2000 }
];

// ==================== DOM ELEMENTS ====================
const screens = {
    home: document.getElementById('screen-home'),
    game: document.getElementById('screen-game'),
    results: document.getElementById('screen-results')
};

// ==================== SCREEN MANAGEMENT ====================
function showScreen(screenName) {
    Object.values(screens).forEach(s => {
        s.classList.remove('active');
        setTimeout(() => {
            if (!s.classList.contains('active')) s.style.display = 'none';
        }, 400);
    });
    const target = screens[screenName];
    target.style.display = 'flex';
    setTimeout(() => target.classList.add('active'), 10);
}

// ==================== HOME SCREEN ====================
function updateHomeStats() {
    const levelInfo = levels[Math.min(gameState.level - 1, levels.length - 1)];
    document.getElementById('home-level').textContent = `Nivel ${gameState.level}: ${levelInfo.name}`;
    document.getElementById('home-xp').textContent = gameState.xp;
    document.getElementById('home-xp-next').textContent = gameState.xpToNext;
}

document.getElementById('config-form').addEventListener('submit', function(e) {
    e.preventDefault();
    gameState.config.focus = document.getElementById('focus-select').value;
    gameState.config.area = document.getElementById('area-select').value;
    gameState.config.questionCount = parseInt(document.querySelector('input[name="question-count"]:checked').value);
    gameState.config.timerLimit = parseInt(document.querySelector('input[name="timer-limit"]:checked').value);
    startGame();
});

// ==================== GAME LOGIC ====================
function startGame() {
    gameState.currentQuestionIndex = 0;
    gameState.score = 0;
    gameState.wrongAnswers = [];
    gameState.answered = false;
    const shuffled = [...mockQuestions].sort(() => Math.random() - 0.5);
    gameState.currentQuestions = shuffled.slice(0, gameState.config.questionCount);
    document.getElementById('game-level-text').textContent = `Nivel ${gameState.level}: ${levels[Math.min(gameState.level - 1, levels.length - 1)].name}`;
    showScreen('game');
    loadQuestion();
}

function loadQuestion() {
    const question = gameState.currentQuestions[gameState.currentQuestionIndex];
    gameState.answered = false;
    const progress = ((gameState.currentQuestionIndex) / gameState.currentQuestions.length) * 100;
    document.getElementById('progress-bar').style.width = progress + '%';
    document.getElementById('progress-text').textContent = `${gameState.currentQuestionIndex + 1} / ${gameState.currentQuestions.length}`;
    document.getElementById('score-display').textContent = gameState.score;
    document.getElementById('question-number').textContent = gameState.currentQuestionIndex + 1;
    document.getElementById('question-category').textContent = question.category;
    document.getElementById('question-text').textContent = question.text;

    const optionsContainer = document.getElementById('options-container');
    optionsContainer.innerHTML = '';
    question.options.forEach((opt, index) => {
        const btn = document.createElement('button');
        btn.className = 'option-btn w-full text-left p-4 md:p-5 rounded-2xl border-2 border-slate-700 bg-slate-800/60 text-white font-medium text-sm md:text-base hover:border-violet-500/50 hover:bg-slate-800 transition-all';
        btn.innerHTML = `<span class="inline-block w-8 h-8 rounded-lg bg-slate-700 text-center leading-8 mr-3 text-sm font-bold text-slate-300">${String.fromCharCode(65 + index)}</span>${opt}`;
        btn.onclick = () => selectAnswer(index);
        optionsContainer.appendChild(btn);
    });

    document.getElementById('feedback-area').classList.add('hidden');
    setupTimer();
}

function setupTimer() {
    const timerContainer = document.getElementById('timer-container');
    if (gameState.config.timerLimit > 0) {
        timerContainer.classList.remove('hidden');
        gameState.timeLeft = gameState.config.timerLimit;
        updateTimerDisplay();
        if (gameState.timerInterval) clearInterval(gameState.timerInterval);
        gameState.timerInterval = setInterval(() => {
            gameState.timeLeft--;
            updateTimerDisplay();
            if (gameState.timeLeft <= 0) {
                clearInterval(gameState.timerInterval);
                timeUp();
            }
        }, 1000);
    } else {
        timerContainer.classList.add('hidden');
    }
}

function updateTimerDisplay() {
    const timerBar = document.getElementById('timer-bar');
    const timerText = document.getElementById('timer-seconds');
    const timerContainer = document.getElementById('timer-container');
    const percentage = (gameState.timeLeft / gameState.config.timerLimit) * 100;
    timerBar.style.width = percentage + '%';
    timerText.textContent = gameState.timeLeft + 's';

    if (percentage > 50) {
        timerBar.className = 'timer-bar h-full bg-gradient-to-r from-emerald-500 to-emerald-400';
        timerText.className = 'font-mono font-bold text-emerald-400';
        timerContainer.classList.remove('timer-critical');
    } else if (percentage > 25) {
        timerBar.className = 'timer-bar h-full bg-gradient-to-r from-amber-500 to-amber-400';
        timerText.className = 'font-mono font-bold text-amber-400';
    } else {
        timerBar.className = 'timer-bar h-full bg-gradient-to-r from-red-500 to-red-400';
        timerText.className = 'font-mono font-bold text-red-400';
        timerContainer.classList.add('timer-critical');
    }
}

function timeUp() {
    if (gameState.answered) return;
    const question = gameState.currentQuestions[gameState.currentQuestionIndex];
    const buttons = document.querySelectorAll('.option-btn');
    gameState.wrongAnswers.push({
        question: question,
        selected: -1,
        reason: "Tempo esgotado!"
    });
    buttons.forEach((btn, idx) => {
        btn.disabled = true;
        if (idx === question.correct) {
            btn.classList.add('reveal-correct');
        } else {
            btn.style.opacity = '0.5';
        }
    });
    showFeedback(false, "Tempo esgotado!");
}

function selectAnswer(selectedIndex) {
    if (gameState.answered) return;
    gameState.answered = true;
    if (gameState.timerInterval) {
        clearInterval(gameState.timerInterval);
        gameState.timerInterval = null;
    }
    const question = gameState.currentQuestions[gameState.currentQuestionIndex];
    const buttons = document.querySelectorAll('.option-btn');
    const isCorrect = selectedIndex === question.correct;
    buttons.forEach(btn => btn.disabled = true);

    if (isCorrect) {
        gameState.score++;
        buttons[selectedIndex].classList.add('correct');
        showFeedback(true, "Resposta correta! Excelente!");
        createConfetti();
    } else {
        buttons[selectedIndex].classList.add('wrong');
        buttons[question.correct].classList.add('reveal-correct');
        gameState.wrongAnswers.push({
            question: question,
            selected: selectedIndex,
            reason: question.explanation
        });
        showFeedback(false, "Resposta incorreta. Nao desista!");
    }
}

function showFeedback(isCorrect, message) {
    const feedbackArea = document.getElementById('feedback-area');
    const feedbackMessage = document.getElementById('feedback-message');
    feedbackArea.classList.remove('hidden');
    feedbackMessage.className = `text-center mb-4 p-4 rounded-2xl font-semibold ${isCorrect ? 'bg-emerald-500/20 text-emerald-300 border border-emerald-500/30' : 'bg-red-500/20 text-red-300 border border-red-500/30'}`;
    feedbackMessage.textContent = message;
    document.getElementById('score-display').textContent = gameState.score;
}

document.getElementById('next-btn').addEventListener('click', function() {
    gameState.currentQuestionIndex++;
    if (gameState.currentQuestionIndex >= gameState.currentQuestions.length) {
        showResults();
    } else {
        loadQuestion();
    }
});

// ==================== RESULTS ====================
function showResults() {
    showScreen('results');
    const total = gameState.currentQuestions.length;
    const correct = gameState.score;
    const wrong = total - correct;
    const percentage = Math.round((correct / total) * 100);

    document.getElementById('result-correct').textContent = correct;
    document.getElementById('result-wrong').textContent = wrong;
    document.getElementById('result-percentage').textContent = percentage + '%';

    const circle = document.getElementById('score-circle');
    const circumference = 2 * Math.PI * 45;
    const offset = circumference - (percentage / 100) * circumference;
    setTimeout(() => {
        circle.style.strokeDashoffset = offset;
    }, 300);

    const xpGained = correct * 10;
    const oldLevel = gameState.level;
    gameState.xp += xpGained;

    let leveledUp = false;
    let leveledDown = false;

    while (gameState.level < levels.length && gameState.xp >= gameState.xpToNext) {
        gameState.xp -= gameState.xpToNext;
        gameState.level++;
        gameState.xpToNext = levels[Math.min(gameState.level - 1, levels.length - 1)].xpNeeded;
        leveledUp = true;
    }

    if (percentage < 50 && gameState.level > 1) {
        gameState.level--;
        gameState.xpToNext = levels[Math.min(gameState.level - 1, levels.length - 1)].xpNeeded;
        gameState.xp = Math.floor(gameState.xpToNext * 0.3);
        leveledDown = true;
    }

    if (percentage < 50 && !leveledDown) {
        gameState.xp = Math.max(0, gameState.xp - 30);
    }

    document.getElementById('result-xp-gained').textContent = (percentage < 50 ? '' : '+') + (percentage < 50 ? -30 : xpGained);
    document.getElementById('result-xp-gained').className = 'text-2xl font-bold ' + (percentage < 50 ? 'text-red-400' : 'text-emerald-400');
    document.getElementById('result-total-xp').textContent = gameState.xp;

    const currentLevelInfo = levels[Math.min(gameState.level - 1, levels.length - 1)];
    document.getElementById('result-level-text').textContent = `Nivel ${gameState.level}: ${currentLevelInfo.name}`;
    document.getElementById('result-xp-current').textContent = gameState.xp;
    document.getElementById('result-xp-next').textContent = gameState.xpToNext;
    const xpPercentage = (gameState.xp / gameState.xpToNext) * 100;
    setTimeout(() => {
        document.getElementById('result-xp-bar').style.width = xpPercentage + '%';
    }, 500);

    const levelMsgDiv = document.getElementById('level-message');
    levelMsgDiv.classList.remove('hidden');

    if (leveledUp) {
        levelMsgDiv.className = 'text-center p-4 rounded-2xl mb-4 bg-gradient-to-r from-violet-500/20 to-purple-500/20 border border-violet-500/40 text-violet-200 font-bold animate-slide-up';
        levelMsgDiv.innerHTML = `🎉 SUBIU DE NIVEL! Agora voce e <span class="text-white">Nivel ${gameState.level}: ${currentLevelInfo.name}</span>! Continue assim!`;
        createConfetti();
        document.getElementById('result-icon').textContent = '🚀';
        document.getElementById('result-title').textContent = 'Nivel Alcancado!';
    } else if (leveledDown) {
        levelMsgDiv.className = 'text-center p-4 rounded-2xl mb-4 bg-gradient-to-r from-red-500/20 to-orange-500/20 border border-red-500/40 text-red-200 font-bold animate-slide-up';
        levelMsgDiv.innerHTML = `⚠️ Voce se afastou do seu objetivo, revise seus pontos fracos! Voltou para <span class="text-white">Nivel ${gameState.level}</span>. Nao desista!`;
        document.getElementById('result-icon').textContent = '💪';
        document.getElementById('result-title').textContent = 'Continue Estudando!';
    } else if (percentage >= 70) {
        levelMsgDiv.className = 'text-center p-4 rounded-2xl mb-4 bg-gradient-to-r from-emerald-500/20 to-teal-500/20 border border-emerald-500/40 text-emerald-200 font-bold animate-slide-up';

        document.getElementById('result-icon').textContent = '🏆';
        document.getElementById('result-title').textContent = 'Missao Concluida!';
        createConfetti();

        levelMsgDiv.className = 'text-center p-4 rounded-2xl mb-4 bg-gradient-to-r from-amber-500/20 to-yellow-500/20 border border-amber-500/40 text-amber-200 font-bold animate-slide-up';

                        <p class="text-xs text-slate-400 leading-relaxed bg-slate-900/50 rounded-lg p-3 border border-slate-700/50">💡 <span class="text-violet-300 font-semibold">Oraculo diz:</span> ${wa.reason}</p>
                    </div>
                </div>
            `;
            oracleList.appendChild(item);
        });
    } else {
        oracleSection.classList.add('hidden');
    }

    updateHomeStats();
}

function createConfetti() {
    const colors = ['#7c3aed', '#a855f7', '#ec4899', '#10b981', '#f59e0b', '#3b82f6'];
    for (let i = 0; i < 50; i++) {
        const confetti = document.createElement('div');
        confetti.className = 'confetti';
        confetti.style.left = Math.random() * 100 + 'vw';
        confetti.style.backgroundColor = colors[Math.floor(Math.random() * colors.length)];
        confetti.style.width = Math.random() * 8 + 4 + 'px';
        confetti.style.height = Math.random() * 8 + 4 + 'px';
        confetti.style.borderRadius = Math.random() > 0.5 ? '50%' : '0';
        confetti.style.animationDuration = Math.random() * 2 + 2 + 's';
        confetti.style.animationDelay = Math.random() * 0.5 + 's';
        document.body.appendChild(confetti);
        setTimeout(() => confetti.remove(), 4000);
    }
}
